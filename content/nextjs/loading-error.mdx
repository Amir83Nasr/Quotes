## Loading.tsx

Create `loading.tsx` next to `page.tsx` to show an immediate loading state:

```tsx
// app/dashboard/loading.tsx
export default function DashboardLoading() {
  return (
    <div className="space-y-4">
      <div className="h-8 bg-gray-200 rounded animate-pulse w-48" />
      <div className="grid gap-4 md:grid-cols-3">
        {Array.from({ length: 3 }).map((_, i) => (
          <div key={i} className="h-32 bg-gray-200 rounded animate-pulse" />
        ))}
      </div>
    </div>
  )
}
```

## Error.tsx

Errors are caught by error boundaries. The `error.tsx` file lets the user retry:

```tsx
"use client"

export default function DashboardError({
  error,
  reset,
}: {
  error: Error & { digest?: string }
  reset: () => void
}) {
  return (
    <div className="p-6 text-center">
      <h2 className="text-xl font-bold text-red-600">Something went wrong</h2>
      <p className="text-gray-600 mt-2">{error.message}</p>
      <button
        onClick={() => reset()}
        className="mt-4 px-4 py-2 bg-blue-600 text-white rounded"
      >
        Try again
      </button>
    </div>
  )
}
```

<Alert>
  <AlertTitle>Error Boundaries Are Client Components</AlertTitle>
  <AlertDescription>
    error.tsx must be a Client Component ("use client"). The error boundary only catches errors in its segment and below — not in the layout that wraps it.
  </AlertDescription>
</Alert>

## Global Error

For root-level layout errors, use `global-error.tsx`:

```tsx
"use client"

export default function GlobalError({
  error,
  reset,
}: {
  error: Error & { digest?: string }
  reset: () => void
}) {
  return (
    <html>
      <body>
        <h1>Critical Error</h1>
        <button onClick={() => reset()}>Reload</button>
      </body>
    </html>
  )
}
```

## Suspense Within Pages

Use React Suspense for streaming parts of a page independently:

```tsx
import { Suspense } from "react"
import { PostSkeleton } from "./skeletons"

export default function BlogPage() {
  return (
    <div>
      <h1>Blog</h1>
      <Suspense fallback={<PostSkeleton />}>
        <BlogPosts />
      </Suspense>
    </div>
  )
}

async function BlogPosts() {
  const posts = await fetch("https://api.example.com/posts").then((r) => r.json())
  return <ul>{/* render posts */}</ul>
}
```

  solutionCode={`"use client"
import { useState, useEffect } from "react"

type User = { id: string; name: string; email: string }

export default function UserProfile({ userId }: { userId: string }) {
  const [user, setUser] = useState<User | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  async function fetchUser() {
    setLoading(true)
    setError(null)
    try {
      const res = await fetch(\`/api/users/\${userId}\`)
      if (!res.ok) throw new Error("Failed to fetch user")
      const data = await res.json()
      setUser(data)
    } catch (err) {
      setError(err instanceof Error ? err.message : "Unknown error")
    } finally {
      setLoading(false)
    }
  }

  useEffect(() => { fetchUser() }, [userId])

  if (loading) return <div className="space-y-2 animate-pulse"><div className="h-6 bg-gray-200 rounded w-48" /><div className="h-4 bg-gray-200 rounded w-64" /></div>
  if (error) return <div className="text-red-600 p-4 border rounded"><p>{error}</p><button onClick={fetchUser} className="mt-2 px-3 py-1 bg-blue-600 text-white rounded">Retry</button></div>

  return <div><h2>{user!.name}</h2><p>{user!.email}</p></div>
}`}
  hints={["Track three states: loading, error, data", "Use try/catch in the fetch function", "animate-pulse is Tailwind's skeleton animation utility"]}
