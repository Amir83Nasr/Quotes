## Loading.tsx

Create `loading.tsx` next to `page.tsx` to show an immediate loading state:

```tsx
// app/dashboard/loading.tsx
export default function DashboardLoading() {
  return (
    <div className="space-y-4">
      <div className="h-8 w-48 animate-pulse rounded bg-gray-200" />
      <div className="grid gap-4 md:grid-cols-3">
        {Array.from({ length: 3 }).map((_, i) => (
          <div key={i} className="h-32 animate-pulse rounded bg-gray-200" />
        ))}
      </div>
    </div>
  )
}
```

## Error.tsx

Errors are caught by error boundaries. The `error.tsx` file lets the user retry:

```tsx
"use client"

export default function DashboardError({
  error,
  reset,
}: {
  error: Error & { digest?: string }
  reset: () => void
}) {
  return (
    <div className="p-6 text-center">
      <h2 className="text-xl font-bold text-red-600">Something went wrong</h2>
      <p className="mt-2 text-gray-600">{error.message}</p>
      <button
        onClick={() => reset()}
        className="mt-4 rounded bg-blue-600 px-4 py-2 text-white"
      >
        Try again
      </button>
    </div>
  )
}
```

## Global Error

For root-level layout errors, use `global-error.tsx`:

```tsx
"use client"

export default function GlobalError({
  error,
  reset,
}: {
  error: Error & { digest?: string }
  reset: () => void
}) {
  return (
    <html>
      <body>
        <h1>Critical Error</h1>
        <button onClick={() => reset()}>Reload</button>
      </body>
    </html>
  )
}
```

## Suspense Within Pages

Use React Suspense for streaming parts of a page independently:

```tsx
import { Suspense } from "react"
import { PostSkeleton } from "./skeletons"

export default function BlogPage() {
  return (
    <div>
      <h1>Blog</h1>
      <Suspense fallback={<PostSkeleton />}>
        <BlogPosts />
      </Suspense>
    </div>
  )
}

async function BlogPosts() {
  const posts = await fetch("https://api.example.com/posts").then((r) =>
    r.json()
  )
  return <ul>{/* render posts */}</ul>
}
```

## See It In Action

<LessonCodeExample
  title="Page with Loading Skeleton + Error Boundary"
  code={`"use client"
import { useState, useEffect } from "react"

function Skeleton() {
return (
<div className="space-y-3 animate-pulse">
<div className="h-8 w-48 bg-gray-200 rounded" />
<div className="h-4 w-64 bg-gray-200 rounded" />
<div className="h-4 w-32 bg-gray-200 rounded" />
<div className="h-24 w-full bg-gray-200 rounded" />
</div>
)
}

export default function UserProfile({ userId }: { userId: string }) {
  const [user, setUser] = useState<{ name: string; email: string; bio: string } | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

async function fetchUser() {
setLoading(true)
setError(null)
try {
const res = await fetch("https://jsonplaceholder.typicode.com/users/" + userId)
if (!res.ok) throw new Error("Failed to fetch user")
const data = await res.json()
setUser({ name: data.name, email: data.email, bio: data.company?.catchPhrase ?? "" })
} catch (err) {
setError(err instanceof Error ? err.message : "Something went wrong")
} finally {
setLoading(false)
}
}

useEffect(() => { fetchUser() }, [userId])

if (loading) return <Skeleton />
if (error)
return (
<div className="p-4 text-center border border-red-200 rounded-lg">
<p className="text-red-600 font-medium">{error}</p>
<button onClick={fetchUser} className="mt-3 px-4 py-2 bg-blue-600 text-white rounded">
Retry
</button>
</div>
)
if (!user) return <p className="text-gray-500">No user data.</p>

return (
<div className="space-y-2 p-4 border rounded-lg">
<h2 className="text-xl font-bold">{user.name}</h2>
<p className="text-gray-600">{user.email}</p>
<p className="text-gray-700">{user.bio}</p>
</div>
)
}`}
explanation="Demonstrates three UI states in a single client component: a skeleton placeholder renders while data loads, user details display on success, and an error state with a retry button handles fetch failures. Uses useEffect with a dependency on userId to re-fetch when the prop changes."
previewType="mixed"
/>
