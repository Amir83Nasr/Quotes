## Named Exports

```javascript
// math.js
export const PI = 3.14159
export function add(a, b) {
  return a + b
}
// app.js
import { PI, add } from "./math.js"
import * as math from "./math.js"
import { add as sum } from "./math.js"
```

## Default Exports

One per module. No name required at import.

```javascript
// logger.js
export default function log(msg) {
  console.log("[LOG]", msg)
}
// app.js
import log from "./logger.js" // any name works
import User, { ROLES } from "./user.js" // mix default + named
```

## Dynamic import()

Load on demand — code splitting.

```javascript
// Static: bundled eagerly
import { format } from "date-fns"
// Dynamic: code-split
if (isAdmin) {
  const admin = await import("./admin-panel.js")
  admin.init()
}
```

## Module Resolution

```javascript
import { useEffect } from "react" // bare specifier → node_modules
import { helper } from "./utils.js" // relative — extension required
```

## Tree-Shaking

Bundlers remove unused exports traced from static imports.

```javascript
// ✅ Named exports: tree-shakeable
export function used() {}
export function unused() {} // eliminated
// ❌ export default { a, b } — hard to tree-shake
```

<Alert>
  <AlertTriangle />
  <AlertTitle>Anti-Pattern</AlertTitle>
  <AlertDescription>
    `export * from "./utils"` (barrel files) kills tree-shaking. Direct imports
    are safer.
  </AlertDescription>
</Alert>

## CommonJS vs ESM


  </TabsContent>

  <TabsContent value="e">

```javascript
import { PI } from "./math.js"

```

  </TabsContent>
</Tabs>

| Feature         | CJS             | ESM               |
| --------------- | --------------- | ----------------- |
| Syntax          | require/exports | import/export     |
| Loading         | Sync, runtime   | Async, parse-time |
| Tree-shaking    | No              | Yes               |
| Top-level await | No              | Yes               |

## See It In Action

<LessonCodeExample
  title="Refactor into Modules"
  code={`// ---- lib/api.ts (module) ----
export const API_URL = "https://api.example.com"

export function handleError(e: unknown) {
  console.error(e instanceof Error ? e.message : String(e))
}

export async function fetchUser(id: string) {
  try {
    const r = await fetch(API_URL + '/users/' + id)
    if (!r.ok) throw Error('HTTP ' + r.status)
    return await r.json()
  } catch (e) {
    handleError(e)
    return null
  }
}

// ---- app/user.ts (importing module) ----
import { fetchUser } from "./lib/api"

export async function renderUser(id: string) {
  const u = await fetchUser(id)
  if (!u) return '<p>User not found</p>'
  return '<h1>' + u.name + '</h1>'
}`}
  language="typescript"
  explanation="Modules separate concerns: `lib/api.ts` owns data fetching, `app/user.ts` owns rendering. Named exports (`export`) and imports (`import { … }`) make dependencies explicit and enable tree-shaking."

}
renderUser(1)`}
