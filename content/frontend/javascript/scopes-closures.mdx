## Scope Types

```javascript
const g = "global"
function outer() {
  const f = "function scope"
  if (true) {
    let b = "block scope"

  }
```

## Hoisting & TDZ

```javascript
console.log(a) // undefined (hoisted)
// console.log(b)       // ReferenceError: TDZ
var a = 1
let b = 2 // hoisted but uninitialized until this line
```

## Closures

A closure remembers its lexical scope even when called elsewhere.

```javascript
function createCounter() {
  let count = 0
  return { increment: () => ++count, getCount: () => count }
}
const c = createCounter()
c.increment() // 1
// No direct access to count

// Private state
function createBankAccount(initial) {
  let balance = initial
  return {
    deposit: (amt) => {
      balance += amt
      return balance
    },
    getBalance: () => balance,
  }
}

// Memoization
function memoize(fn) {
  const cache = new Map()
  return (arg) =>
    cache.has(arg) ? cache.get(arg) : (cache.set(arg, fn(arg)), cache.get(arg))
}
```

## Loop + Closure Gotcha

```javascript
for (var i = 0; i < 3; i++) setTimeout(() => console.log(i), 0) // 3, 3, 3 (shared binding)
for (let i = 0; i < 3; i++) setTimeout(() => console.log(i), 0) // 0, 1, 2 (per-iteration binding)
```

## See It In Action

<LessonCodeExample
  title="Build a Private Counter"
  code={`function createAdvancedCounter(initial = 0) {
  let count = initial
  return {
    increment: (step = 1) => {
      if (step <= 0) throw new Error("Step must be positive")
      count += step
      return count
    },
    decrement: (step = 1) => {
      if (step <= 0) throw new Error("Step must be positive")
      count -= step
      return count
    },
    reset: () => { count = initial; return count },
    getCount: () => count,
  }
}

const counter = createAdvancedCounter(10)
console.log(counter.increment(5)) // 15
console.log(counter.decrement(3)) // 12
console.log(counter.getCount()) // 12
console.log(counter.reset()) // 10`}
  language="javascript"
  explanation="A closure encapsulates `count`— the returned methods close over the variable, making it private. No external code can modify`count`except through`increment`, `decrement`, or `reset`."
/>
