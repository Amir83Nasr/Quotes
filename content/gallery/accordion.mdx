# Accordion

Accordions let users show and hide sections of content.

## Example

<GalleryDemo component="accordionExample" />

## Usage

```tsx
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from "@/components/ui/accordion"

<Accordion type="single" collapsible>
  <AccordionItem value="item-1">
    <AccordionTrigger>Section title</AccordionTrigger>
    <AccordionContent>
      Content inside the accordion panel.
    </AccordionContent>
  </AccordionItem>
</Accordion>
```

## Props

| Prop | Type | Default |
|------|------|---------|
| type | `"single" \| "multiple"` | required |
| collapsible | `boolean` | `false` |
